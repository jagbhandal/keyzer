# KEYZER runtime dependency: PySide6 (Qt 6 for Python).
#
# Supported install path on Ubuntu 26.04 (apt builds it for the system python3,
# so it works on Python 3.14 where PyPI wheels may not yet):
#
#   sudo apt install python3-pyside6.qtquick python3-pyside6.qtquickcontrols2 \
#                    python3-pyside6.qtsvg python3-pyside6.qtdbus
#
# pip alternative (only where PySide6 wheels exist for your Python):
PySide6>=6.6

# Device capture (app/capture.py, tools/probe.py) reads evdev directly. On
# Ubuntu the apt build is python3-evdev; via pip:
evdev>=1.6

# Not Python packages, but required at runtime (install.sh handles these on apt):
#   - input-remapper        the remapping engine (a root systemd service)
#   - x11-xserver-utils     provides `xmodmap`, which the root daemon needs to
#                           resolve key names — without it every keysym binding
#                           is dropped server-side
#   - the Razer udev rule (packaging/70-keyzer.rules), so capture can read the
#     Razer /dev/input nodes without blanket `input`-group access
